CREATE DATABASE bank;

CREATE TABLE account (
	account_number INT,
	holder_name VARCHAR(20),
	account_type VARCHAR(20)
);

CREATE TABLE customer (
	customer_id INT,
	customer_name VARCHAR(20),
	customer_address TEXT
);

CREATE TABLE employee (
	employee_id INT,
	employee_name VARCHAR(20),
	designation TEXT
);
